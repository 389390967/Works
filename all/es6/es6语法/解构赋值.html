  • 		1、数组解构赋值
		{
		    let a, b;
		    [a, b] = [1, 2];
		    console.log(a, b)
		}
		输出：1 2

		{
		    let a, b, rest;
		    [a, b, ...rest] = [1, 2, 3, 4, 5, 6, 7, 8, 9];
		    console.log(a, b, rest)
		}
		输出：1 2 [3, 4, 5, 6, 7, 8, 9]
		
    		2、对象解构赋值
		{
		    let a, b;
		    ({ a, b } = { a: 1, b: 2 });
		    console.log(a, b)
		}
		输出：1 2

		{
			let o = { c: 123, d: 456 };
			let { c, d } = o;
			console.log(c, d)
		}
		输出：123 456
		
    		默认值
		{
		    let a, b, c;
		    [a, b, c =3] = [1, 2];
		    console.log(a, b)
		}
		输出：1 2 3

		{
		    let a, b, c;
		    [a, b, c =3] = [1, 2, 4];
		    console.log(a, b, c)
		}
		输出：1 2 4

		{
			let { a = 1, b = 2 } = { a: 2 };
			console.log(a, b);
		}
		输出：2 2
		
    		应用场景一、变量交换
		{
		    let a = 1,
		        b = 2;
		    [a, b] = [b, a];
		    console.log(a, b)
		}
		输出：2 1
		
    		应用场景二、取返回值
		{
		    function test() {
		        return [1, 2];
		    }
		    let a, b;
		    [a, b] = test();
		    console.log(a, b);
		}
		输出：1 2

		{
		    function test() {
		        return [1, 2, 3, 4];
		    }
		    let a, b;
		    [a, , , b] = test();
		    console.log(a, b);
		}
		输出：1 4

		{
		    function test() {
		        return [1, 2, 3, 4, 5];
		    }
		    let a, b;
		    [a, , ...b] = test();
		    console.log(a, b);
		}
		输出：1 [3, 4, 5]

		{
			let data = {
			    title: "数据标题",
			    desc: [{
			        title: "描述标题",
